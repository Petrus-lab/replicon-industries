  <div class="output" id="output">Awaiting input...</div>

  <script>
    async function sendQuick(action) {
      try {
        const res = await fetch('http://localhost:3001/run', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ action })
        });
        const data = await res.text();
        document.getElementById('output').innerText = data;
      } catch (err) {
        document.getElementById('output').innerText = '‚ùå Runner not responding';
      }
    }

    async function submitToRunner() {
      const actionType = document.getElementById('actionType').value;
      const input = document.getElementById('input').value;

      const responseDisplay = document.getElementById('output');
      responseDisplay.innerText = '‚è≥ Processing...';

      try {
        const res = await fetch('http://localhost:3001/run', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            action: actionType,
            command: input
          })
        });

        const data = await res.json();
        responseDisplay.innerText = data.output || '‚úÖ Command executed with no output.';
      } catch (err) {
        responseDisplay.innerText = '‚ùå Communication failure with runner.';
      }
    }
  </script>
